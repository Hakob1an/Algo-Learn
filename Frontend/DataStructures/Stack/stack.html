class Stack:
    def __init__(self):
        self.items = []

    def push(self, item):
        self.items.append(item)

    def pop(self):
        if len(self.items) == 0:
            return "Stack is empty"
        return self.items.pop()

    def peek(self):
        return self.items[-1] if self.items else "Stack is empty"

    def size(self):
        return len(self.items)
